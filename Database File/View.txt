1.create view Deptartment_Doctor
as
select PERSON_ID,PERSON_ADDRESS,PERSON_NAME,PERSON_PHONE,
DOCTOR_DEPARTMENT,DOCTOR_GENDER,DOCTOR_QUALIFICATION,
DEPARTMENT_ID,CONSULTANT,DEPARTMENT_HEAD from doctor,
department where DOCTOR_DEPARTMENT = DEPARTMENT_NAME;

2.create view labreport_patient
as
select LAB_NO,PATIENT_ID,PERSON_ADDRESS,PERSON_NAME,
PERSON_PHONE,PATIENT_DISEASE,TEST_DATE,PATIENT_DOCTOR,
PATIENT_GENDER from patient,labreport where PATIENT_ID = PERSON_ID;
